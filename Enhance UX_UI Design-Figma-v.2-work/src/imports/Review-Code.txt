        async function main() {
            try {
                await liff.init({ liffId: "2009630242-iPO8WjV7" }); // ใช้ ID เดิมของคุณ
                if (!liff.isLoggedIn()) { liff.login(); return; }

                const profile = await liff.getProfile();
                
                // นำชื่อไปใส่ในช่อง "ชื่อใน LINE"
                document.getElementById("userId").value = profile.userId;
                document.getElementById("displayName").value = profile.displayName;
                
                console.log("ดึงข้อมูลชื่อสำเร็จ: " + profile.displayName);
            } catch (err) {
                console.error("LIFF Error", err);
            }
        }

        // 2. จัดการการส่งข้อมูล
        document.getElementById("regForm").addEventListener("submit", function(e) {
            e.preventDefault();
            const loader = document.getElementById("loader");
            const btn = document.getElementById("submitBtn");
            
            loader.style.display = "flex";
            btn.disabled = true;

            // รวบรวมข้อมูล
            const formData = {
                userId: document.getElementById("userId").value,
                displayName: document.getElementById("displayName").value,
                firstName: document.getElementById("firstName").value,
                lastName: document.getElementById("lastName").value,
                phone: document.getElementById("phone").value,
                petType: Array.from(document.querySelectorAll('input[name="petType"]:checked')).map(el => el.value).join(', '),
                petCount: document.getElementById("petCount").value,
                province: document.getElementById("province").value,
                addressDetail: document.getElementById("addressDetail").value
            };

            google.script.run
                .withSuccessHandler(function(res) {
                    loader.style.display = "none";
                    if (res === "Success") {
                        alert("ยินดีด้วยค่ะ! คุณได้รับ 10 แต้มแรกเข้าเรียบร้อย ✨");
                        // 🚀 ส่งไปหน้าบัตรสมาชิกทันที เพื่อให้ลูกค้าเห็นแต้ม
                        window.location.href = "https://liff.line.me/2009630242-iPO8WjV7"; 
                    } else {
                        alert("เกิดข้อผิดพลาด: " + res);
                        btn.disabled = false;
                    }
                })
                .withFailureHandler(function(err) {
                    loader.style.display = "none";
                    btn.disabled = false;
                    alert("ระบบขัดข้อง: " + err.message);
                })
                .registerNewMember(formData); 
        });

        main();
